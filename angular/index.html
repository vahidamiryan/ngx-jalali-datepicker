Angular ​

Live, interactive examples rendered by the real @vahidamiryan/ngx-jalali-datepicker components, bootstrapped as Angular islands right here on the page. They render the same UI as @vahidamiryan/vue-datepicker from the same @vahidamiryan/datepicker-core, so behavior is identical. Install:

bash
npm install @vahidamiryan/ngx-jalali-datepicker @vahidamiryan/datepicker-core

Setup ​

ts
// app.config.ts
import { provideZonelessChangeDetection } from '@angular/core';
import {
  provideNgxDatepicker,
  JalaliCalendarAdapter,
  GregorianCalendarAdapter,
} from '@vahidamiryan/ngx-jalali-datepicker';

export const appConfig = {
  providers: [
    provideZonelessChangeDetection(),
    provideNgxDatepicker(new JalaliCalendarAdapter(), new GregorianCalendarAdapter('en-US')),
  ],
};
