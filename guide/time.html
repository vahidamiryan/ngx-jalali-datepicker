Headless helpers ​

ts
import { getTimeOfDay, withTimeOfDay } from '@vahidamirian/datepicker-core'

getTimeOfDay(value.start)              // { hours: 14, minutes: 30 }
withTimeOfDay(value.start, 9, 0)       // same day at 09:00 (a new Date)
